## Swarm Masters && DAGs

* <https://www.linkedin.com/pulse/agentic-workflows-why-dag-ultimate-design-pattern-dramil-dodeja-rtkmc/>

  ```
  | $ pip install networkx                            # PRESETUP    
  | 
  | $ cat example_DAG.py --------------------------------------------------------------------------
  | 
  | import networkx as nx
  |  
  | dag = nx.DiGraph()                              # 1. Define a directed graph of tasks
  |  
  | dag.add_nodes_from([                            # 1.2 Add tasks
  |     "set_table", "cook_food",
  |     "prepare_drinks", "serve_dinner" ])
  |  
  | dag.add_edge("set_table", "serve_dinner")       # 1.3 Add dependencies
  | dag.add_edge("cook_food", "serve_dinner")
  | dag.add_edge("prepare_drinks", "serve_dinner")
  |
  | if not nx.is_directed_acyclic_graph(dag):       # 2 Validate DAG (No Cycles)
  |     raise ValueError("Graph contains a cycle")
  | execution_order = list(nx.topological_sort(dag))# 4. Get Execution Order (Topological Sort)
  |  
  | def execute_dag(graph, task_handlers):          # 5. Simple DAG Orchestrator
  |     completed = set()
  |  
  |     for node in nx.topological_sort(graph):
  |         dependencies = set(                     # Check: Ensure all dependencies are complete
  |             graph.predecessors(node))
  |         if not dependencies.issubset(completed):
  |             raise RuntimeError(
  |               f"Dependencies not met for {node}")
  |         task_handlers[node]()
  |         completed.add(node)
  |  
  | def set_table():                                # 6. Task Implementations
  |     print("Table is set")
  | def cook_food():
  |     print("Food is cooked")
  | def prepare_drinks():
  |     print("Drinks are ready")
  | def serve_dinner():
  |     print("Dinner is served")
  |  
  | handlers =
  | execute_dag(dag, {                               # 7. Run the Workflow
  |     "set_table"     : set_table,                 # > Table is set
  |     "cook_food"     : cook_food,                 # > Food is cooked
  |     "prepare_drinks": prepare_drinks,            # > Drinks are ready
  |     "serve_dinner"  : serve_dinner               # > Dinner is served
  | })
  ```
